Sets up a flag with any number of parameters, which can each be of differing types

1) Sets up a flag with no arguments
2) Sets up a flag with one argument
3) Sets up a flag with any number of arguments of the same type
4) Sets up a flag with any number of arguments of any number of types
5) Sets an alias to the previous flag, where this name will be treated as equivalent to the last name by the program

[edit] Template parameters

T - The type of the variable(s)

[edit] Parameters

var - the variable which should be set according to user input
vars - the variables which should be set according to user input, passed as an array of pointers
defaultVal - the default value of the single variable passed
defaultVals - the default values for the passed variables
types - an array of type-constants corresponding to the types in vars, and the way in which defaultVals must be applied to each (if applicable)
argc - the number of elements in vars/defaultVals
indicator - a bool which will be set to true if the flag is set by the user, and false otherwise. Can be passed as a pointer, or by reference.

[edit] Exceptions

None

[edit] Notes

If a flag is set more than once, the last time the flag was set takes precedence

When passing void* arrays to defaultVals, keep in mind that pointers-to-default_values must be passed as pointers-to-pointers-to-those_values.

It is required that the number of elements in each array passed is not less than argc. It is not required that the numbers are equal. Excess elements are ignored (the destructor does not treat excess elements any differently from others).

Currently, when the user manually sets a flag to its default, it is set to said default at that exact moment. This can cause problems with function and pointer defaults, and will likely be changed in the future.

[edit] Example

#include <CIP.h>
#include <string>
using namespace CIP;
 
int main(int argc, char* argv[])
{
    // (setFlag's overloads are called in the order they are listed in the table on this page)

    using namespace TYPE;

    input_handler ih(argc, argv);

    bool indicator[3];
    int orderedVars[3];


    int alpha;
    int beta;
    int gamma;
    CIP_LAMBDA(int, delta) {
        if (argc != 2)
            throw exception();
        return *(int*)argv[0] + *(int*)argv[1];
    };
    int epsilon;


    double zeta[3];

    double eta1;
    double eta2;

    double theta1;
    double theta2;
    double theta3;

    CIP_LAMBDA(double, kappa1) { return 1.1 + *(int*)argv[0]; };
    CIP_LAMBDA(double, kappa2) { return 2.2 + *(int*)argv[0]; };
    CIP_LAMBDA(double, kappa3) { return 3.3 + *(int*)argv[0]; };

    double iota[5];
    double x[6]; // set by something unrelated to inputHandler


    float vnd1; // vnd = void no default
    int vnd2;
    char vnd3;


    std::string someFilename; // set by something unrelated to inputHandler
    std::string v1; // v = void with default
    char v2;
    double v3;

#define elms(x) sizeof(x)/sizeof(x[0]) // number of elements in array x

    CIP_FLAGS_START(ih);
    ih.setFlag("no-args", indicator[0]);
    ih.setFlag("alpha", alpha);
    ih.setFlag("beta", beta, 10);
    ih.setFlag("gamma", gamma, fref(delta, orderedVars[0], orderedVars[1]), indicator[1]);
    ih.setFlag("epsilon", epsilon, &orderedVars[2]);
    ih.setFlag("zeta", arrayVar(zeta), elms(zeta));
    ih.setFlag("eta", tvars(eta1, eta2), defaults(1.1, 2.2), 2);
    ih.setFlag("theta", tvars(theta1, theta2, theta3), defaults(fref(kappa1, orderedVars[2]), fref(kappa1, orderedVars[2]), fref(kappa1, orderedVars[2])), 3);
    ih.setFlag("iota", arrayVar(iota), arrayVar(x), elms(iota));
    ih.setFlag("vnd", vars(vnd1, vnd2, vnd3), types(STRING, CHAR, DOUBLE), 3, indicator[2]);
    ih.setFlag("void", vars(v1, v2, v3), new void*[3]{ NO_FILE_EXTENSION(someFilename), new char{ 'x' }, new double*{&x[5]} }, types(FUNC::STR, CHAR, PTR::DOUBLE), 3);
    ih.setFlag("space");
    CIP_FLAGS_END(ih);

#undef elms
}

Results: 

Flag alpha takes 1 int argument, and does not have a default.
beta takes 1 int argument, whose default value is 10.
gamma takes 1 int argument, whose default value is orderedVar[0] + orderedVar[1].
epsilon takes 1 int argument, whose default value is orderedVar[2]'s value at the time it is set to default.
zeta takes 3 double arguments, and does not have a default.
eta takes 2 double arguments, whose default values are 1.1 and 2.2
theta takes 3 double arguments, whose default values are 1.1 + orderedVars[2], 2.2 + orderedVars[2], and 3.3 + orderedVars[2]
iota takes 5 double arguments, whose default values are the same as the first 5 elements of x
vnd takes 3 arguments of types (string, char, double), and does not have a default
void has an alias "space", takes 3 arguments of types (string, char, double), whose default values are, respectively:
 - the return value of noFileExtension(someFilename)
 - 'x'
 - the value of x[5]
 
 indicator[0] is true if the user set no-args
 indicator[1] is true if the user set gamma
 indicator[2] is true if the user set vnd
